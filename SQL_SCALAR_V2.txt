1)SELECT LOWER(FIRST_NAME), UPPER(LAST_NAME)
FROM EMPLOYEES 
WHERE EMPLOYEE_ID BETWEEN  80 AND 150;

2)SELECT FIRST_NAME, LAST_NAME 
FROM EMPLOYEES 
WHERE LOWER(LAST_NAME)='king';

3)SELECT FIRST_NAME, LAST_NAME, SUBSTR(FIRST_NAME,1,1)||SUBSTR(LAST_NAME,1,3)||'@ORACLE.COM' AS EMAIL_ADD
FROM EMPLOYEES;

4)SELECT FIRST_NAME, LAST_NAME, SUBSTR(FIRST_NAME,1,1)||SUBSTR(LAST_NAME,LENGTH(LAST_NAME)-2,3)||'@ORACLE.COM' AS EMAIL_ADD
FROM EMPLOYEES;

5)SELECT CONCAT(FIRST_NAME,LAST_NAME) 
FROM EMPLOYEES;

6)SELECT LAST_NAME
FROM EMPLOYEES 
WHERE LENGTH(LAST_NAME)>8;

7)SELECT FIRST_NAME, LAST_NAME, PHONE_NUMBER, REPLACE(PHONE_NUMBER,'515','815') AS NEW_PHONE_NUMBER
FROM EMPLOYEES;

8)SELECT FIRST_NAME, LAST_NAME, PHONE_NUMBER, REPLACE(SUBSTR(PHONE_NUMBER,1,3),'515','815')||SUBSTR(PHONE_NUMBER,4) AS NEW_PHONE_NUMBER
FROM EMPLOYEES;

9)SELECT FIRST_NAME, SALARY, SALARY+((12/100)*SALARY), TRUNC(SALARY+((12/100)*SALARY)), ROUND(SALARY+((12/100)*SALARY))
FROM EMPLOYEES;

10)SELECT FIRST_NAME, HIRE_DATE, HIRE_DATE-10, ADD_MONTHS(HIRE_DATE,1), SYSDATE-HIRE_DATE 
FROM EMPLOYEES;

11)SELECT FIRST_NAME, LAST_NAME, HIRE_DATE, TRUNC(MONTHS_BETWEEN(SYSDATE,HIRE_DATE)), TRUNC(MONTHS_BETWEEN(SYSDATE,HIRE_DATE)/12)
FROM EMPLOYEES;

12)SELECT FIRST_NAME, LAST_NAME, HIRE_DATE, EXTRACT(YEAR FROM HIRE_DATE)+1 
FROM EMPLOYEES;

13)SELECT FIRST_NAME, LAST_NAME, HIRE_DATE, ROUND(TO_DATE(HIRE_DATE),'YEAR'), ROUND(TO_DATE(HIRE_DATE),'MONTH') 
FROM EMPLOYEES;SELECT FIRST_NAME, LAST_NAME, SALARY, NVL(TO_CHAR(COMMISSION_PCT),'NO COMMISSION')
FROM EMPLOYEES;

14)SELECT FIRST_NAME, LAST_NAME, HIRE_DATE, EXTRACT(DAY FROM HIRE_DATE), EXTRACT(YEAR FROM HIRE_DATE) 
FROM EMPLOYEES;

15)SELECT UPPER(LAST_NAME), TO_CHAR(SALARY,'9,999.999'), TO_CHAR(HIRE_DATE,'DD/MM/YYYY') 
FROM EMPLOYEES 
WHERE LAST_NAME LIKE 'D%'
OR LAST_NAME LIKE 'K%';

16)SELECT FIRST_NAME, LAST_NAME, SALARY, NVL(COMMISSION_PCT,'0')
FROM EMPLOYEES;

17)SELECT FIRST_NAME, LAST_NAME, SALARY, NVL(TO_CHAR(COMMISSION_PCT),'NO COMMISSION')
FROM EMPLOYEES;

18)SELECT FIRST_NAME, LAST_NAME, SALARY, 
CASE WHEN SALARY BETWEEN 0 AND 5000 THEN 'A'
WHEN SALARY BETWEEN 5001 AND 15000 THEN 'B'
WHEN SALARY BETWEEN 15001 AND 20000 THEN 'C'
ELSE 'D' END AS SALARY_GRADE
FROM EMPLOYEES;
