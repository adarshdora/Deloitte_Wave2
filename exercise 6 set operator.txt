1)SELECT DEPARTMENT_ID
FROM DEPARTMENTS
MINUS
SELECT DEPARTMENT_ID 
FROM EMPLOYEES
WHERE UPPER(JOB_ID)='ST_CLERK';

2)SELECT C.COUNTRY_NAME, C.COUNTRY_ID
FROM COUNTRIES C
MINUS
SELECT C.COUNTRY_NAME, C.COUNTRY_ID
FROM COUNTRIES C
JOIN LOCATIONS L
ON C.COUNTRY_ID=L.COUNTRY_ID
JOIN DEPARTMENTS D
ON L.LOCATION_ID=D.LOCATION_ID;

3)SELECT DISTINCT JOB_ID, DEPARTMENT_ID
FROM EMPLOYEES
WHERE DEPARTMENT_ID=10
UNION ALL
SELECT DISTINCT JOB_ID, DEPARTMENT_ID
FROM EMPLOYEES
WHERE DEPARTMENT_ID=50
UNION ALL 
SELECT DISTINCT JOB_ID, DEPARTMENT_ID
FROM EMPLOYEES
WHERE DEPARTMENT_ID=20;

4)SELECT EMPLOYEE_ID, JOB_ID
FROM EMPLOYEES
INTERSECT
SELECT EMPLOYEE_ID, JOB_ID
FROM JOB_HISTORY;

5)SELECT LAST_NAME, DEPARTMENT_ID, TO_CHAR(NULL)
FROM EMPLOYEES
UNION
SELECT TO_CHAR(NULL), DEPARTMENT_ID,DEPARTMENT_NAME
FROM DEPARTMENTS;
