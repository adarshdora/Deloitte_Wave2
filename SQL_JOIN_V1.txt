1)SELECT C.FIRST_NAME, C.LAST_NAME, C.PACK_ID, P.SPEED
FROM CUSTOMERS C, PACKAGES P
WHERE C.PACK_ID=P.PACK_ID; 

2)SELECT C.FIRST_NAME, C.LAST_NAME, C.PACK_ID, P.SPEED
FROM CUSTOMERS C, PACKAGES P
WHERE C.PACK_ID IN (22,27)
AND C.PACK_ID=P.PACK_ID
ORDER BY C.LAST_NAME;

3)SELECT P.PACK_ID, P.SPEED, P.MONTHLY_PAYMENT, S.SECTOR_NAME
FROM PACKAGES P, SECTORS S
WHERE P.SECTOR_ID=S.SECTOR_ID;

4)SELECT C.FIRST_NAME, C.LAST_NAME, C.PACK_ID, P.SPEED, P.MONTHLY_PAYMENT, S.SECTOR_NAME
FROM CUSTOMERS C, PACKAGES P, SECTORS S
WHERE C.PACK_ID=P.PACK_ID AND P.SECTOR_ID=S.SECTOR_ID;

5)SELECT C.FIRST_NAME, C.LAST_NAME, C.PACK_ID, P.SPEED, P.MONTHLY_PAYMENT, S.SECTOR_NAME
FROM CUSTOMERS C, PACKAGES P, SECTORS S
WHERE C.PACK_ID=P.PACK_ID AND P.SECTOR_ID=S.SECTOR_ID AND S.SECTOR_NAME='Business';

6)SELECT C.LAST_NAME, C.FIRST_NAME, C.JOIN_DATE, C.PACK_ID, P.SPEED, S.SECTOR_NAME 
FROM CUSTOMERS C, PACKAGES P, SECTORS S
WHERE C.PACK_ID=P.PACK_ID AND P.SECTOR_ID=S.SECTOR_ID AND S.SECTOR_NAME='Private' AND EXTRACT(YEAR FROM C.JOIN_DATE)=2006;

7)SELECT P.PACK_ID, P.SPEED, P.MONTHLY_PAYMENT, PG.GRADE_NAME
FROM PACKAGES P, PACK_GRADES PG
WHERE P.PACK_ID=PG.GRADE_ID;

8)SELECT C.FIRST_NAME, C.LAST_NAME, P.SPEED, P.MONTHLY_PAYMENT
FROM CUSTOMERS C, PACKAGES P
WHERE C.PACK_ID=P.PACK_ID;

9)SELECT C.FIRST_NAME, C.LAST_NAME, P.SPEED, P.MONTHLY_PAYMENT
FROM CUSTOMERS C, PACKAGES P
WHERE C.PACK_ID=P.PACK_ID(+);

10)SELECT C.FIRST_NAME, C.LAST_NAME, P.SPEED, P.MONTHLY_PAYMENT
FROM CUSTOMERS C, PACKAGES P
WHERE C.PACK_ID(+)=P.PACK_ID;

11)SELECT C.FIRST_NAME, C.LAST_NAME, P.SPEED, P.MONTHLY_PAYMENT
FROM CUSTOMERS C
FULL JOIN PACKAGES P
ON C.PACK_ID=P.PACK_ID;

12)SELECT C1.LAST_NAME, C1.FIRST_NAME, C1.PACK_ID
FROM CUSTOMERS C1, CUSTOMERS C2
WHERE C1.PACK_ID=C2.PACK_ID AND C2.FIRST_NAME='Amado' AND C2.LAST_NAME='Taylor';

13)SELECT C1.LAST_NAME, C1.FIRST_NAME, C1.MONTHLY_DISCOUNT
FROM CUSTOMERS C1, CUSTOMERS C2
WHERE C1.PACK_ID=C2.PACK_ID AND C1.MONTHLY_DISCOUNT<C2.MONTHLY_DISCOUNT AND C2.CUSTOMER_ID=103;

14)SELECT P.PACK_ID, P.SPEED
FROM PACKAGES P, PACKAGES P1
WHERE P.PACK_ID=P1.PACK_ID AND P.SPEED=P1.SPEED AND P1.PACK_ID=10;
